#include <stdio.h>
#include <math.h>

int ilerozw(float a, float b, float c, float *rozw1, float *rozw2)
{
float delta, pdelta;
delta=b*b-4*a*c;
pdelta=sqrt(delta);
if(delta>0)
  {
  *rozw1=(-b-pdelta)/(2*a);
  *rozw2=(-b+pdelta)/(2*a);
  return 2;
  }
if(delta==0)
  {
  *rozw1=-b/(2*a);
  return 1;
  }
else
  {
  return 0;
  }
}

int main()
{
float a,b,c,rozw1,rozw2;
int rozw;

rozw=ilerozw(a,b,c,&rozw1,&rozw2);
if(rozw==2)
